define the feature functions¶

  • feature vector is defined by $(1, f_1(x, y), f_2(x, y), \cdots, f_{k-1}(x, y)) \in \mathbb{R}^k$
In [ ]:
def compute_feature(point):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    one_vector = np.ones(len(point))
    x = point[:,0]
    y = point[:,1]
    feature = np.column_stack((one_vector, one_vector, np.power(x, 2), y)) 
    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return feature

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature = $(1, f_1(x, y), \cdots, f_{k-1}(x, y)) \in \mathbb{R}^k$

def compute_linear_regression(theta, feature):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    # print("feature shape: ", feature.shape)
    # print("theta shape: ", theta.shape)
    features = feature * theta
    term2 = features[:,1] * np.sin(features[:,2])
    # print("term2 shape: ", term2.shape)
    value = features[:,0] + term2 + features[:,-1]
    # print("value shape: ", value.shape)
    
    # value = feature@theta

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    
    return value

define the logistic regression function¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature $= (1, f_1(x, y), \cdots, f_{k-1}(x, y) \in \mathbb{R}^k$
In [ ]:
def compute_logistic_regression(theta, feature):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    value = sigmoid(compute_linear_regression(theta, feature))

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return value

define the residual function¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature $= (1, f_1(x, y), \cdots, f_{k-1}(x, y) \in \mathbb{R}^k$
  • label $= l \in \{0, 1\}^k$
In [ ]:
def compute_residual(theta, feature, label):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    h = compute_logistic_regression(theta, feature) 
    term1 = label*np.log(h)
    term2 = (1 - label)*np.log(1 - h)
    residual = (-1 * term1) - term2

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return residual

define the loss function for the logistic regression¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature $= (1, f_1(x, y), \cdots, f_{k-1}(x, y) \in \mathbb{R}^k$
  • label $= l \in \{0, 1\}^k$
In [ ]:
def compute_loss(theta, feature, label, alpha):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #

    n = len(feature)
    residual = compute_residual(theta, feature, label) 

    l_term = (alpha * theta@theta)/2
    loss = np.sum(residual)/n + l_term
    #regularisation
    

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return loss

define the gradient of the loss with respect to the model parameter $\theta$¶

  • $\theta = (\theta_0, \theta_1, \cdots, \theta_{k-1}) \in \mathbb{R}^k$
  • feature $= (1, f_1(x, y), \cdots, f_{k-1}(x, y) \in \mathbb{R}^k$
  • label $= l \in \{0, 1\}^k$

def compute_gradient(theta, feature, label, alpha):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    h = compute_logistic_regression(theta, feature)
    y = label
    x = feature
    gradient = np.sum(x.T*(h - y), -1) / len(x) + alpha*theta #음수?

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return gradient

compute the accuracy of the prediction for point with a given model parameter¶

In [ ]:
def compute_accuracy(theta, feature, label):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    length = len(label)
    prediction = (compute_logistic_regression(theta, feature) >= 0.5).astype(int)
    correct = (prediction == label).sum()
    accuracy = (np.sum(correct) / length) * 100
    #error check

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    return accuracy

accuracy_iteration_train    = np.zeros(number_iteration)
accuracy_iteration_test     = np.zeros(number_iteration)

run the gradient descent algorithm to optimize the loss function with respect to the model parameter¶

In [ ]:
for i in range(number_iteration):

    # ++++++++++++++++++++++++++++++++++++++++++++++++++
    # complete the blanks
    #
    train_feature = compute_feature(data_train_point)
    test_feature = compute_feature(data_test_point)

    train_label = data_train_label
    test_label = data_test_label

    theta           = theta - learning_rate * compute_gradient(theta, train_feature, train_label, alpha)
    loss_train      = compute_loss(theta, train_feature, train_label, alpha)
    loss_test       = compute_loss(theta, test_feature, test_label, alpha)
    accuracy_train  = compute_accuracy(theta, train_feature, train_label)
    accuracy_test   = compute_accuracy(theta, test_feature, test_label)

    #
    # ++++++++++++++++++++++++++++++++++++++++++++++++++

    loss_iteration_train[i]     = loss_train
    loss_iteration_test[i]      = loss_test
    accuracy_iteration_train[i] = accuracy_train
    accuracy_iteration_test[i]  = accuracy_test

theta_optimal = theta
